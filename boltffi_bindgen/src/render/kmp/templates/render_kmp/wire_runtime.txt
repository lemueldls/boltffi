class FfiException(val code: Int, message: String) : Exception(message)

private inline fun <T> useWireBytes(bytes: ByteArray, block: (WireReader) -> T): T {
    val borrowed = WireReaderPool.acquire(bytes)
    return try {
        block(borrowed.reader)
    } finally {
        borrowed.close()
    }
}

sealed class BoltFFIResult<out T, out E> {
    data class Ok<T>(val value: T) : BoltFFIResult<T, Nothing>()
    data class Err<E>(val error: E) : BoltFFIResult<Nothing, E>()

    val isSuccess: Boolean get() = this is Ok
    val isFailure: Boolean get() = this is Err
}

object Utf8Codec {
    fun maxBytes(value: String): Int = value.length * 3
}

interface WireCodec<T> {
    fun decode(reader: WireReader): T
    fun wireEncodedSize(value: T): Int
    fun wireEncodeTo(wire: WireWriter, value: T)
}

class WireReader(private var buffer: ByteArray) {
    internal fun reset(next: ByteArray) {
        buffer = next
        pos = 0
    }

    private var pos: Int = 0

    private fun leI16(offset: Int): Short =
        ((buffer[offset].toInt() and 0xFF) or ((buffer[offset + 1].toInt() and 0xFF) shl 8)).toShort()

    private fun leI32(offset: Int): Int =
        (buffer[offset].toInt() and 0xFF) or
            ((buffer[offset + 1].toInt() and 0xFF) shl 8) or
            ((buffer[offset + 2].toInt() and 0xFF) shl 16) or
            ((buffer[offset + 3].toInt() and 0xFF) shl 24)

    private fun leI64(offset: Int): Long =
        (buffer[offset].toLong() and 0xFFL) or
            ((buffer[offset + 1].toLong() and 0xFFL) shl 8) or
            ((buffer[offset + 2].toLong() and 0xFFL) shl 16) or
            ((buffer[offset + 3].toLong() and 0xFFL) shl 24) or
            ((buffer[offset + 4].toLong() and 0xFFL) shl 32) or
            ((buffer[offset + 5].toLong() and 0xFFL) shl 40) or
            ((buffer[offset + 6].toLong() and 0xFFL) shl 48) or
            ((buffer[offset + 7].toLong() and 0xFFL) shl 56)

    fun readBool(): Boolean {
        val value = buffer[pos] != 0.toByte()
        pos += 1
        return value
    }

    fun readI8(): Byte {
        val value = buffer[pos]
        pos += 1
        return value
    }

    fun readU8(): UByte {
        val value = buffer[pos].toUByte()
        pos += 1
        return value
    }

    fun readI16(): Short {
        val value = leI16(pos)
        pos += 2
        return value
    }

    fun readU16(): UShort {
        val value = leI16(pos).toUShort()
        pos += 2
        return value
    }

    fun readI32(): Int {
        val value = leI32(pos)
        pos += 4
        return value
    }

    fun readU32(): UInt {
        val value = leI32(pos).toUInt()
        pos += 4
        return value
    }

    fun readI64(): Long {
        val value = leI64(pos)
        pos += 8
        return value
    }

    fun readU64(): ULong {
        val value = leI64(pos).toULong()
        pos += 8
        return value
    }

    fun readF32(): Float {
        val value = Float.fromBits(leI32(pos))
        pos += 4
        return value
    }

    fun readF64(): Double {
        val value = Double.fromBits(leI64(pos))
        pos += 8
        return value
    }

    fun readString(): String {
        val len = readU32().toInt()
        if (len == 0) return ""
        val start = pos
        pos += len
        return buffer.decodeToString(start, start + len)
    }

    fun readBytes(): ByteArray {
        val len = readU32().toInt()
        if (len == 0) return ByteArray(0)
        val start = pos
        pos += len
        return buffer.copyOfRange(start, start + len)
    }

    inline fun <T> readOption(reader: (WireReader) -> T): T? {
        val tag = readU8().toInt()
        return if (tag == 0) null else reader(this)
    }

    inline fun <T> readOptional(reader: (WireReader) -> T): T? = readOption(reader)

    inline fun <T> readList(reader: (WireReader) -> T): List<T> {
        val count = readU32().toInt()
        if (count == 0) return emptyList()
        return ArrayList<T>(count).also { list -> repeat(count) { list.add(reader(this)) } }
    }

    inline fun <T, E> readResult(
        okReader: (WireReader) -> T,
        errReader: (WireReader) -> E,
    ): BoltFFIResult<T, E> {
        val tag = readU8().toInt()
        return if (tag == 0) {
            BoltFFIResult.Ok(okReader(this))
        } else {
            BoltFFIResult.Err(errReader(this))
        }
    }

    fun skip(count: Int) {
        pos += count
    }

    internal val backingBuffer: ByteArray
        get() = buffer
}

class WireWriter(initialCapacity: Int = 256) {
    private var buffer: ByteArray = ByteArray(initialCapacity.coerceAtLeast(0))
    private var pos: Int = 0

    internal fun reset(requiredCapacity: Int) {
        if (requiredCapacity < 0) {
            throw IllegalArgumentException("requiredCapacity must be >= 0")
        }

        if (buffer.size < requiredCapacity) {
            buffer = ByteArray(requiredCapacity)
        }
        pos = 0
    }

    internal val capacity: Int
        get() = buffer.size

    private fun ensureCapacity(needed: Int) {
        val required = pos + needed
        if (required <= buffer.size) return

        val nextSize = maxOf(buffer.size * 2, required)
        buffer = buffer.copyOf(nextSize)
    }

    private fun writeBoolean(offset: Int, value: Boolean) {
        buffer[offset] = if (value) 1 else 0
    }

    private fun putI8(offset: Int, value: Byte) {
        buffer[offset] = value
    }

    private fun putI16(offset: Int, value: Short) {
        buffer[offset] = (value.toInt() and 0xFF).toByte()
        buffer[offset + 1] = ((value.toInt() ushr 8) and 0xFF).toByte()
    }

    private fun putI32(offset: Int, value: Int) {
        buffer[offset] = (value and 0xFF).toByte()
        buffer[offset + 1] = ((value ushr 8) and 0xFF).toByte()
        buffer[offset + 2] = ((value ushr 16) and 0xFF).toByte()
        buffer[offset + 3] = ((value ushr 24) and 0xFF).toByte()
    }

    private fun putI64(offset: Int, value: Long) {
        buffer[offset] = (value and 0xFFL).toByte()
        buffer[offset + 1] = ((value ushr 8) and 0xFFL).toByte()
        buffer[offset + 2] = ((value ushr 16) and 0xFFL).toByte()
        buffer[offset + 3] = ((value ushr 24) and 0xFFL).toByte()
        buffer[offset + 4] = ((value ushr 32) and 0xFFL).toByte()
        buffer[offset + 5] = ((value ushr 40) and 0xFFL).toByte()
        buffer[offset + 6] = ((value ushr 48) and 0xFFL).toByte()
        buffer[offset + 7] = ((value ushr 56) and 0xFFL).toByte()
    }

    fun writeBool(value: Boolean) {
        ensureCapacity(1)
        writeBoolean(pos, value)
        pos += 1
    }

    fun writeI8(value: Byte) {
        ensureCapacity(1)
        putI8(pos, value)
        pos += 1
    }

    fun writeU8(value: UByte) = writeI8(value.toByte())

    fun writeI16(value: Short) {
        ensureCapacity(2)
        putI16(pos, value)
        pos += 2
    }

    fun writeU16(value: UShort) = writeI16(value.toShort())

    fun writeI32(value: Int) {
        ensureCapacity(4)
        putI32(pos, value)
        pos += 4
    }

    fun writeU32(value: UInt) = writeI32(value.toInt())

    fun writeI64(value: Long) {
        ensureCapacity(8)
        putI64(pos, value)
        pos += 8
    }

    fun writeU64(value: ULong) = writeI64(value.toLong())
    fun writeF32(value: Float) = writeI32(value.toBits())
    fun writeF64(value: Double) = writeI64(value.toBits())

    fun writePadding(count: Int) {
        if (count <= 0) return
        ensureCapacity(count)
        buffer.fill(0, pos, pos + count)
        pos += count
    }

    fun writeString(value: String) {
        val bytes = value.encodeToByteArray()
        writeU32(bytes.size.toUInt())
        ensureCapacity(bytes.size)
        bytes.copyInto(buffer, destinationOffset = pos)
        pos += bytes.size
    }

    fun writeBytes(value: ByteArray) {
        writeU32(value.size.toUInt())
        ensureCapacity(value.size)
        value.copyInto(buffer, destinationOffset = pos)
        pos += value.size
    }

    inline fun <T> writeOption(value: T?, writer: (WireWriter, T) -> Unit) {
        if (value == null) {
            writeU8(0u)
            return
        }

        writeU8(1u)
        writer(this, value)
    }

    inline fun <T> writeOptional(value: T?, writer: (WireWriter, T) -> Unit) =
        writeOption(value, writer)

    inline fun <T> writeList(values: List<T>, writer: (WireWriter, T) -> Unit) {
        writeU32(values.size.toUInt())
        values.forEach { writer(this, it) }
    }

    inline fun <T, E> writeResult(
        value: BoltFFIResult<T, E>,
        okWriter: (WireWriter, T) -> Unit,
        errWriter: (WireWriter, E) -> Unit,
    ) {
        when (value) {
            is BoltFFIResult.Ok -> {
                writeU8(0u)
                okWriter(this, value.value)
            }

            is BoltFFIResult.Err -> {
                writeU8(1u)
                errWriter(this, value.error)
            }
        }
    }

    fun writePrimitiveList(v: IntArray) {
        writeU32(v.size.toUInt())
        v.forEach(::writeI32)
    }

    fun writePrimitiveList(v: LongArray) {
        writeU32(v.size.toUInt())
        v.forEach(::writeI64)
    }

    fun writePrimitiveList(v: FloatArray) {
        writeU32(v.size.toUInt())
        v.forEach(::writeF32)
    }

    fun writePrimitiveList(v: DoubleArray) {
        writeU32(v.size.toUInt())
        v.forEach(::writeF64)
    }

    fun writePrimitiveList(v: ShortArray) {
        writeU32(v.size.toUInt())
        v.forEach(::writeI16)
    }

    fun writePrimitiveList(v: ByteArray) {
        writeU32(v.size.toUInt())
        ensureCapacity(v.size)
        v.copyInto(buffer, destinationOffset = pos)
        pos += v.size
    }

    fun writePrimitiveList(v: BooleanArray) {
        writeU32(v.size.toUInt())
        v.forEach(::writeBool)
    }

    @JvmName("writeIntList")
    fun writePrimitiveList(v: List<Int>) {
        writeU32(v.size.toUInt())
        v.forEach(::writeI32)
    }

    @JvmName("writeLongList")
    fun writePrimitiveList(v: List<Long>) {
        writeU32(v.size.toUInt())
        v.forEach(::writeI64)
    }

    fun toByteArray(): ByteArray = buffer.copyOf(pos)

    fun toByteArray(size: Int): ByteArray {
        if (size < 0 || size > buffer.size) {
            throw IllegalArgumentException("size must be in 0..capacity")
        }
        return buffer.copyOf(size)
    }
}

private const val MAX_CACHED_WIRE_BUFFER_BYTES: Int = 1024 * 1024

class WireReaderPoolState(private val cacheSize: Int = 4) {
    private val cachedReaders: Array<WireReader?> = arrayOfNulls(cacheSize)
    private var depth: Int = 0

    fun acquire(buffer: ByteArray): BorrowedWireReader {
        val slot = depth
        depth = slot + 1
        val shouldCache = buffer.size <= MAX_CACHED_WIRE_BUFFER_BYTES && slot < cacheSize
        val reader = if (shouldCache) {
            cachedReaders[slot] ?: WireReader(buffer).also { cachedReaders[slot] = it }
        } else {
            WireReader(buffer)
        }
        reader.reset(buffer)
        return BorrowedWireReader(this, reader)
    }

    fun release() {
        depth -= 1
    }
}

class BorrowedWireReader(
    private val state: WireReaderPoolState,
    val reader: WireReader,
) {
    internal val buffer: ByteArray
        get() = reader.backingBuffer

    fun close() {
        state.release()
    }
}

object WireReaderPool {
    private val state = WireReaderPoolState()

    fun acquire(buffer: ByteArray): BorrowedWireReader = state.acquire(buffer)
}

class WireWriterPoolState(private val cacheSize: Int = 4) {
    private val cachedWriters: Array<WireWriter?> = arrayOfNulls(cacheSize)
    private var depth: Int = 0

    fun acquire(requiredCapacity: Int): BorrowedWireWriter {
        val slot = depth
        depth = slot + 1
        val shouldCache = requiredCapacity <= MAX_CACHED_WIRE_BUFFER_BYTES && slot < cacheSize
        val writer = if (shouldCache) {
            cachedWriters[slot] ?: WireWriter(requiredCapacity).also { cachedWriters[slot] = it }
        } else {
            WireWriter(requiredCapacity)
        }
        writer.reset(requiredCapacity)
        return BorrowedWireWriter(this, writer)
    }

    fun release() {
        depth -= 1
    }
}

class BorrowedWireWriter(
    private val state: WireWriterPoolState,
    val writer: WireWriter,
) {
    internal val buffer: ByteArray
        get() = writer.toByteArray()

    fun toByteArray(): ByteArray = writer.toByteArray()

    fun close() {
        state.release()
    }
}

object WireWriterPool {
    private val state = WireWriterPoolState()

    fun acquire(requiredCapacity: Int): BorrowedWireWriter = state.acquire(requiredCapacity)
}
